def create
  asset = TinymceAsset.new
  asset.file_type = params[:file].content_type
  asset.file = params[:file]
  asset.save!


  render json: {
    image: {
      # url: serve_tinymce_asset_path(asset)
      url: asset.file.url
    }
  }, content_type: "text/html"
end
