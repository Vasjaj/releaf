def initialize resource, field_name_prefix
  @resource = resource
  @klass = @resource.class
  @field_name_prefix = field_name_prefix
  @errors = {}
  @error_message_i18n_scope = "activerecord.errors.messages.#{@klass.name.underscore}"
  format_errors
end
